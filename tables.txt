CREATE TABLE countries (
    id      VARCHAR(10) PRIMARY KEY,
    name_ja VARCHAR(100) NOT NULL,
    lat     FLOAT NOT NULL,
    lng     FLOAT NOT NULL
);

CREATE TABLE trade_links (
    id         SERIAL PRIMARY KEY,
    from_country VARCHAR(10) REFERENCES countries(id),
    to_country   VARCHAR(10) REFERENCES countries(id),
    value        FLOAT NOT NULL,
    category     VARCHAR(50) NOT NULL,
    year         INTEGER NOT NULL
);

CREATE TABLE diplomatic_relations (
    id             SERIAL PRIMARY KEY,
    country_a      VARCHAR(10) REFERENCES countries(id),
    country_b      VARCHAR(10) REFERENCES countries(id),
    relation_type  VARCHAR(50),
    summary        TEXT,
    source_url     TEXT,
    analyzed_at    TIMESTAMP DEFAULT NOW()
);

CREATE TABLE users (
    id            SERIAL PRIMARY KEY,
    email         VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    display_name  VARCHAR(100) NOT NULL,
    role          VARCHAR(20) NOT NULL DEFAULT 'editor',
    created_at    TIMESTAMP DEFAULT NOW()
);
CREATE TABLE diplomatic_proposals (
    id            SERIAL PRIMARY KEY,
    country_a     VARCHAR(10) REFERENCES countries(id),
    country_b     VARCHAR(10) REFERENCES countries(id),
    relation_type VARCHAR(50) NOT NULL,
    summary       TEXT NOT NULL,
    source_url    TEXT NOT NULL,
    source_note   TEXT,
    status        VARCHAR(20) NOT NULL DEFAULT 'pending',
    proposed_by   INTEGER REFERENCES users(id),
    reviewed_by   INTEGER REFERENCES users(id),
    review_comment TEXT,
    created_at    TIMESTAMP DEFAULT NOW(),
    reviewed_at   TIMESTAMP
);
CREATE TABLE edit_history (
    id          SERIAL PRIMARY KEY,
    relation_id INTEGER REFERENCES diplomatic_relations(id),
    changed_by  INTEGER REFERENCES users(id),
    before_data JSONB,
    after_data  JSONB,
    comment     TEXT,
    changed_at  TIMESTAMP DEFAULT NOW()
);
